00552   void adaptC2(const int hsig)
00553   {
00554     const int N = sp.N;
00555     bool diag = sp.diagonalCov == 1 || sp.diagonalCov >= gen;
00556 
00557     if(sp.ccov != T(0))
00558     {
00559       // definitions for speeding up inner-most loop
00560       const T mucovinv = T(1)/sp.mucov;
00561       const T commonFactor = sp.ccov * (diag ? (N + T(1.5)) / T(3) : T(1));
00562       const T ccov1 = std::min(commonFactor*mucovinv, T(1));
00563       const T ccovmu = std::min(commonFactor*(T(1)-mucovinv), T(1)-ccov1);
00564       const T sigmasquare = sigma*sigma;
00565       const T onemccov1ccovmu = T(1)-ccov1-ccovmu;
00566       const T longFactor = (T(1)-hsig)*sp.ccumcov*(T(2)-sp.ccumcov);
00567 
00568       eigensysIsUptodate = false;
00569 
00570       // update covariance matrix
00571       for(int i = 0; i < N; ++i)
00572         for(int j = diag ? i : 0; j <= i; ++j)
00573         {
00574           T& Cij = C[i][j];
00575           Cij = onemccov1ccovmu*Cij + ccov1 * (rgpc[i]*rgpc[j] + longFactor*Cij);
00576           for(int k = 0; k < sp.mu; ++k)
00577           { // additional rank mu update
00578             const T* rgrgxindexk = rgrgx[index[k]];
00579             Cij += ccovmu*sp.weights[k] * (rgrgxindexk[i] - rgxold[i])
00580                 * (rgrgxindexk[j] - rgxold[j]) / sigmasquare;
00581           }
00582         }
00583       // update maximal and minimal diagonal value
00584       maxdiagC = mindiagC = C[0][0];
00585       for(int i = 1; i < N; ++i)
00586       {
00587         const T& Cii = C[i][i];
00588         if(maxdiagC < Cii)
00589           maxdiagC = Cii;
00590         else if(mindiagC > Cii)
00591           mindiagC = Cii;
00592       }
00593     }

01115   T* const* samplePopulation()
01116   {
01117     bool diag = sp.diagonalCov == 1 || sp.diagonalCov >= gen;
01118 
01119     // calculate eigensystem
01120     if(!eigensysIsUptodate)

01138     { // generate scaled random vector D*z

01140       for(int i = 0; i < sp.N; ++i)
01141         if(diag)
01142           rgrgxink[i] = xmean[i] + sigma*rgD[i]*rand.gauss();
01143         else

01147         {

01149           for(int j = 0; j < sp.N; ++j)
01150             sum += B[i][j]*rgdTmp[j];

01154

01248   T* updateDistribution(const T* fitnessValues)
01249   {
01250     const int N = sp.N;
01251     bool diag = sp.diagonalCov == 1 || sp.diagonalCov >= gen;
01252 
01253     assert(state != UPDATED && "updateDistribution(): You need to call "
01254           "samplePopulation() before update can take place.");
01255     assert(fitnessValues && "updateDistribution(): No fitness function value array input.");
01256 
01257     if(state == SAMPLED) // function values are delivered here
01258       countevals += sp.lambda;
01259     else
01260       assert(false && "updateDistribution(): unexpected state");
01261 
01262     // assign function values
01263     for(int i = 0; i < sp.lambda; ++i)
01264       rgrgx[i][N] = rgFuncValue[i] = fitnessValues[i];
01265 
01266     // Generate index
01267     sortIndex(fitnessValues, index, sp.lambda);
01268 
01269     // Test if function values are identical, escape flat fitness
01270     if(fitnessValues[index[0]] == fitnessValues[index[(int) sp.lambda / 2]])
01271     {
01272       sigma *= std::exp(T(0.2) + sp.cs / sp.damps);
01273       assert(false && "Warning: sigma increased due to equal function values\n"
01274           "   Reconsider the formulation of the objective function");
01275     }
01276 
01277     // update function value history
01278     for(int i = (int) *(arFuncValueHist - 1) - 1; i > 0; --i)
01279       arFuncValueHist[i] = arFuncValueHist[i - 1];
01280     arFuncValueHist[0] = fitnessValues[index[0]];
01281 
01282     // update xbestever
01283     if(rgxbestever[N] > rgrgx[index[0]][N] || gen == 1)
01284       for(int i = 0; i <= N; ++i)
01285       {
01286         rgxbestever[i] = rgrgx[index[0]][i];
01287         rgxbestever[N+1] = countevals;
01288       }
01289 
01290     const T sqrtmueffdivsigma = std::sqrt(sp.mueff) / sigma;
01291     // calculate xmean and rgBDz~N(0,C)
01292     for(int i = 0; i < N; ++i)
01293     {
01294       rgxold[i] = xmean[i];
01295       xmean[i] = 0.;
01296       for(int iNk = 0; iNk < sp.mu; ++iNk)
01297         xmean[i] += sp.weights[iNk]*rgrgx[index[iNk]][i];
01298       rgBDz[i] = sqrtmueffdivsigma*(xmean[i]-rgxold[i]);
01299     }
01300 
01301     // calculate z := D^(-1)* B^(-1)* rgBDz into rgdTmp
01302     for(int i = 0; i < N; ++i)
01303     {
01304       T sum;
01305       if(diag)
01306         sum = rgBDz[i];
01307       else
01308       {
01309         sum = 0.;
01310         for(int j = 0; j < N; ++j)
01311           sum += B[j][i]*rgBDz[j];
01312       }
01313       rgdTmp[i] = sum/rgD[i];
01314     }
01315 
01316     // cumulation for sigma (ps) using B*z
01317     const T sqrtFactor = std::sqrt(sp.cs*(T(2)-sp.cs));
01318     const T invps = T(1)-sp.cs;
01319     for(int i = 0; i < N; ++i)
01320     {
01321       T sum;
01322       if(diag)
01323         sum = rgdTmp[i];
01324       else
01325       {
01326         sum = T(0);
01327         T* Bi = B[i];
01328         for(int j = 0; j < N; ++j)
01329           sum += Bi[j]*rgdTmp[j];
01330       }
01331       rgps[i] = invps*rgps[i] + sqrtFactor*sum;
01332     }
01333 
01334     // calculate norm(ps)^2
01335     T psxps(0);
01336     for(int i = 0; i < N; ++i)
01337     {
01338       const T& rgpsi = rgps[i];
01339       psxps += rgpsi*rgpsi;
01340     }
01341 
01342     // cumulation for covariance matrix (pc) using B*D*z~N(0,C)
01343     int hsig = std::sqrt(psxps) / std::sqrt(T(1) - std::pow(T(1) - sp.cs, T(2)* gen))
01344         / chiN < T(1.4) + T(2) / (N + 1);
01345     const T ccumcovinv = 1.-sp.ccumcov;
01346     const T hsigFactor = hsig*std::sqrt(sp.ccumcov*(T(2)-sp.ccumcov));
01347     for(int i = 0; i < N; ++i)
01348       rgpc[i] = ccumcovinv*rgpc[i] + hsigFactor*rgBDz[i];
01349 
01350     // update of C
01351     adaptC2(hsig);
01352 
01353     // update of sigma
01354     sigma *= std::exp(((std::sqrt(psxps) / chiN) - T(1))* sp.cs / sp.damps);
01355 
01356     state = UPDATED;
01357     return xmean;
01358   }
